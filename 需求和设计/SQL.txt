中文表
ch: {
ch_id INT PRIMARY KEY,
ch_text TEXT,
trans VARCHAR,
author VARCHAR,
publisher VARCHAR,
pub_date VARCHAR
}

日文表
jp: {
jp_id INT PRIMARY KEY,
jp_text TEXT,
trans VARCHAR,
author VARCHAR,
publisher VARCHAR,
pub_date
}

日文中文连接表
ch_jp: {
user_num INT,
ch_num INT CONSTRAINT chfk FOREIGN KEY (ch_id) REFERENCES ch(ch_id) ON DELETE SET NULL ON UPDATE CASCADE,
jp_num INT CONSTRAINT jpfk FOREIGN KEY (jp_id) REFERENCES jp(jp_id) ON DELETE SET NULL ON UPDATE CASCADE,
}
